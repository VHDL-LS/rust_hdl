unary_operator ::= "??" | "+" | "-" | "abs" | "not" | "and" | "or" | "nand" | "nor" | "xor" | "xnor"

unbounded_array_definition ::= "array" "(" [ index_subtype_definition_list ] ")" "of" subtype_indication

unit_declarations ::= "units" primary_unit_declaration { secondary_unit_declaration }

use_clause ::= "use" [ name_list ] ";"

use_clause_context_item ::= use_clause

use_clause_declaration ::= use_clause

variable_assignment_statement ::= simple_variable_assignment | conditional_variable_assignment | selected_variable_assignment

variable_declaration ::= "variable" identifier_list ":" subtype_indication [ ":=" ] [ expression ] ";"

verification_unit_binding_indication ::= "use" "vunit" [ verification_unit_list ]

verification_unit_list ::= name { "," name }

wait_statement ::= [ label ] "wait" [ sensitivity_clause ] [ condition_clause ] [ timeout_clause ] ";"

waveform ::= waveform_elements | unaffected_waveform

waveform_element ::= expression [ "after" expression ]

waveform_elements ::= waveform_element { "," waveform_element }

while_iteration_scheme ::= "while" expression
